# Array operations

<CodeDivWrapper>

```js
const ascorder = (a, b) => a - b;
[2, 2, 1]
  .sort(ascorder) // [1,2,2]

  [(1, 2, 3)].reverse() // [3,2,1]
  [(1, 2, 3)].map((val) => val * 2) // [2,4,6]
  [(1, 2, 3)].filter((val) => val < 2) // [1]

  [(1, 2, 3)].flatMap((val) => [val, val * 2]) // [1, 2, 2, 4, 3, 6]
  [([1, 2], [2, 3])].flatMap((val) => val); // [1, 2, 2, 3]
const a = [1, [2, 3], [4, 5, [6, 7]]];
a.flat(); // [1, 2, 3, 4 , 5, [2,6]]
a.flat().flat(); // [1, 2, 3, 4, 5, 2, 6]
a.flat(Infinity); // [1, 2, 3, 4, 5, 2, 6]

const reducer = (accumulator, currentValue) => accumulator + currentValue;

[1, 2, 3]
  .reduce(reducer) // 6

  [(1, 2, 3)].reduce(reducer, 4) // 4 + 1 + 2 + 3 = 10

  [(1, 2, 1, 1, 2)].lastIndexOf(2) // 4
  [(1, 2, 1, 1, 2)].indexOf(2) // 1

  [(1, 2, 1, 1, 2)].join("_") // "1_2_1_1_2"

  [(1, 2)].concat([3]); // [1, 2, 3]

for (let val of [3, 4, 5]) {
  console.log(val);
} // 3 4 5
```

</CodeDivWrapper>
