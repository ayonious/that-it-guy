### Step1:

Add a tsconfig.json file

<CodeDivWrapper>

```json
{
  "include": ["./src/*", "test"],
  "compilerOptions": {
    "lib": ["dom", "es2015"],
    "jsx": "react",
    "target": "es5",
    "allowJs": true,
    "skipLibCheck": true,
    "esModuleInterop": true,
    "allowSyntheticDefaultImports": true,
    "strict": true,
    "forceConsistentCasingInFileNames": true,
    "module": "esnext",
    "moduleResolution": "node",
    "resolveJsonModule": true,
    "isolatedModules": true,
    "noEmit": true
  }
}
```

</CodeDivWrapper>
